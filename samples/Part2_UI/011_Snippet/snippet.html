$(function(){
  // Sliderウィジェットを適用
  $('#slider').slider({
    range: true,  // 範囲スライダーを有効化
    values: [100, 400], // 初期値（範囲）
    min: 0, // 最小値
    max: 500, // 最大値
    // スライダーの値が変化した時の処理
    change: function (event, ui) {
      // テキストボックスにスライダーの値を反映
      $('#value1').val(ui.values[0]);
      $('#value2').val(ui.values[1]);
    }
  });
});
